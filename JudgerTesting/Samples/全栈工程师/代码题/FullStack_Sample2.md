模拟候选人口述思路：

并查集的 findRoot 核心就是沿着 parent 数组往上找，直到找到一个节点的 parent 指向自身，那就是根节点。路径压缩的优化思想是：在查找过程中，把沿途经过的每个节点的 parent 都直接指向根节点，这样下次再查这些节点时就是 O(1) 直达。我用递归来实现，先递归找到根，回溯时顺便把当前节点的 parent 更新为根。时间均摊接近 O(1)，不需要额外空间。


模拟候选人代码：

function findRoot(x, parent) {
    // 如果 x 的父节点就是自身，说明 x 是根节点
    if (parent[x] === x) {
        return x;
    }
    // 递归查找根节点，并将路径上的节点直接挂到根节点下
    parent[x] = findRoot(parent[x], parent);
    return parent[x];
}


备注：标准满分回答。语法正确可运行，路径压缩功能完全实现，口述思路与代码一致。时间均摊 O(α(n)) 接近 O(1) 最优，空间 O(1)（递归栈深度经路径压缩后极浅）最优，变量命名清晰（x/parent/findRoot），关键步骤有注释。

function findRoot(a, b) {
    // 拷贝一份 parent 数组避免中间状态污染
    let c = [...b];

    function d(e, f) {
        if (f[e] === e) {
            return e;

模拟候选人代码：

function findRoot(a, b) {
    let c = [...b];
    let d = [];
    let e = a;
    while (c[e] !== e) {
        d.push(e);
        e = c[e];
    }
    for (let f = 0; f < d.length; f++) {
        c[d[f]] = e;
    }
    b.forEach((_, i) => b[i] = c[i]);
    return e;
}
